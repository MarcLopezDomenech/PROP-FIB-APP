107  public boolean showLoader(JFrame reference) {
108  LoaderDialog dialog = new LoaderDialog();
109  Pair<String, String[]> languageAndPaths = dialog.initialize(reference);
110 
111  if (languageAndPaths == null) return false; // No s'ha introduït cap path
112 
113  String language = languageAndPaths.getFirst();
114  String[] paths = languageAndPaths.getSecond();
115  for (String path : paths) {
116  try {
117  // Importem el document i aconseguim les dades (fav, títol, autor)
118  cp.importDocument(path, language);
119  } catch (ExceptionInvalidFormat | ExceptionDocumentExists | ExceptionInvalidCharacter | ExceptionMissingTitleOrAuthor e) {
120  // Excepcions nostres que indiquen errors que es poden donar
121  showError(reference, e.getMessage());
122  } catch (FileNotFoundException e) {
123  // Excepció de Java produïda quan la ruta no és correcta
124  showError(reference, "La ruta que has especificat no es correcta");
125  } catch (ExceptionInvalidLanguage e) {
126  // Per presentació garantim que això no pot passar. Si es dona, tenim error intern
127  showInternalError(reference);
128  }
129  }
130 
131  return true;
132  }
