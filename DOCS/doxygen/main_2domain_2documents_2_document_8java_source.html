312  private double termRelevance_tf_idf(String[] terms, Integer num_docs, Map<String,Integer> presence) {
313  double tf_idf = 0;
314  double totalTerms = (double) internalDoc.getTotalWords();
315  Map<String,Integer> relevantTerms = internalDoc.getRelevantWords();
316  for (String term : terms) {
317  if (relevantTerms.containsKey(term)) {
318  tf_idf += (relevantTerms.get(term)/totalTerms) * Math.log10((double) num_docs/presence.get(term));
