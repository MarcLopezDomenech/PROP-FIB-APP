207  private void onOK() {
208  String content_fin = textcont.getText();
209  err = false;
210  try {
211  if (!Objects.equals(cont, content_fin)) {
212  cp.updateContentDocument(tit, auth, content_fin);
213  }
214  if (!Objects.equals(lang, setlang)) {
215  cp.updateLanguageDocument(tit, auth, setlang);
216  }
217  } catch (ExceptionNoDocument e) {
218  cvd.showError(modify, "No existeix el document 2");
219  err = true;
220  } catch (ExceptionInvalidLanguage e) {
221  cvd.showError(modify, "No existeix la llengua");
222  err = true;
223  }
224  if (!err) {
225  try {
226  if (!Objects.equals(tit, tit_field.getText()) || !Objects.equals(auth, aut_field.getText())) {
227  String nout = tit_field.getText();
228  String nouat = aut_field.getText();
229  cp.updateTitleAndAuthorDocument(tit, auth, nout, nouat);
230  }
231  } catch (ExceptionNoDocument e) {
232  cvd.showError(modify, "No existeix el document");
233  err = true;
234  } catch (ExceptionDocumentExists e) {
235  cvd.showError(modify, "Ja existeix un document amb aquest títol i autor");
236  err = true;
237  }
238  }
239  if (!err) {
240  lang = setlang;
241  cont = content_fin;
242  tit = tit_field.getText();
243  auth = aut_field.getText();
244  buttonOK.setEnabled(false);
245  }
