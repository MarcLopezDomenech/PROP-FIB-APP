284  public List<Pair<String, String>> listSimilars(String title, String author, int k, String strategy) throws ExceptionNoDocument, ExceptionInvalidStrategy, ExceptionInvalidK {
285  // Comprovem que l'estratègia és vàlida
286  if (!"tf-idf".equals(strategy) && !"tf-boolean".equals(strategy)) throw new ExceptionInvalidStrategy(strategy);
287 
288  // Comprovem que el valor de k és vàlid
289  if (k < 0) throw new ExceptionInvalidK(k);
290 
291  Document original = getDocument(title, author);
292  List<Pair<Pair<String, String>, Double>> ordre = new ArrayList<>();
293  for (Map.Entry<String, Map<String, Document>> authorTitleDoc : documents.entrySet()) {
294  String aut = authorTitleDoc.getKey();
295  Map<String, Document> titleDocs = authorTitleDoc.getValue();
296  for (Map.Entry<String, Document> titleDoc : titleDocs.entrySet()) {
297  String tit = titleDoc.getKey();
298  // En cas que no estiguem en el document original
299  if (!(tit.equals(title) && aut.equals(author))) {
300  Document doc = titleDoc.getValue();
301  // Obtenim el valor (en funció de l'estratègia triada) i l'afegim a la llista
302  Double value;
303  if ("tf-idf".equals(strategy)) value = original.compare_tf_idf(doc, numDocuments, presence);
304  else value = original.compare_tf_boolean(doc);
305  ordre.add(new Pair<>(new Pair<>(tit, aut), value));
306  }
307  }
308  }
309  // Ordenem la llista en funció dels valors
310  Collections.sort(ordre, new ValueComparator());
311  List<Pair<String, String>> result = new ArrayList<>();
312  Iterator<Pair<Pair<String, String>, Double>> iterator = ordre.listIterator();
313  int i = 0;
314  // Iterem per tota la llista ordenada fins que arribem a k o al final, per guardar el resultat que retornem
315  while(i < k && iterator.hasNext()) {
316  result.add(iterator.next().getFirst());
317  ++i;
318  }
